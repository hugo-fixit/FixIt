{{- /* Build and minify a JS resource with unified behavior. @param {resource.Resource} .Resource Hugo resource to process @param {Object|Boolean} [.Build] true for defaults, dict for custom options, false/omit to skip js.Build @param {Boolean} [.Minify] extra minify flag; final minify is Minify OR Build.minify @return {resource.Resource} */ -}} {{- $resource := .Resource -}} {{- $minify := .Minify -}} {{- if .Build -}} {{- $buildOptions := .Build -}} {{- if eq .Build true -}} {{- $buildOptions = dict -}} {{- end -}} {{- if not (isset $buildOptions "targetPath") -}} {{- $targetPath := $resource.Name | replaceRE `\.ts$` ".js" -}} {{- $buildOptions = dict "targetPath" $targetPath | merge $buildOptions -}} {{- end -}} {{- if not (isset $buildOptions "sourceMap") -}} {{- $buildOptions = dict "sourceMap" (cond hugo.IsDevelopment "linked" "none") | merge $buildOptions -}} {{- end -}} {{- if not (isset $buildOptions "minify") -}} {{- $buildOptions = dict "minify" hugo.IsProduction | merge $buildOptions -}} {{- end -}} {{- $minify = $minify | or (index $buildOptions "minify") -}} {{- $resource = $resource | js.Build $buildOptions -}} {{- end -}} {{- if $minify -}} {{- $resource = $resource | minify -}} {{- end -}} {{- return $resource -}}
