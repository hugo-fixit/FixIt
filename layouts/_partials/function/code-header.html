` (lower .Type) $titleEl $ellipsisBtn $editBtn $copyBtn -}} {{- if strings.Contains $wrapper `` -}} {{- $wrapper = replaceRE `(

$1
` $codeHeader) $wrapper -}} {{- end -}} {{- return $wrapper -}}
